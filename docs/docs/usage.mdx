# Usage

Import `TrueSheet`.

```tsx
import { TrueSheet } from "@lodev09/react-native-true-sheet"
```

Define `TrueSheet` inside any component and attach a `ref` to it.

```tsx {2-2,10-12}
const App = () => {
  const sheet = useRef<TrueSheet>(null)

  // add present here
  // add dismiss here

  return (
    <View>
      <Button onPress={present} title="Open Sheet" />
      <TrueSheet ref={sheet} sizes={['auto', '80%', 'large']}>
        <Button onPress={dismiss} title="Dismiss" />
      </TrueSheet>
    </View>
  )
}
```

Use the `ref` instance in presenting the sheet.

```tsx {2}
const present = async () => {
  await sheet.current?.present()
  console.log('horray! sheet has been presented ğŸ’©')
}
```

The same goes for dismissing it.

```tsx {2}
const dismiss = async () => {
  await sheet.current?.dismiss()
  console.log('Bye bye ğŸ‘‹')
}
```
